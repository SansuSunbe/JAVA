
- switch 문
	```
	switch(변수){
		case 값1 : 변수 == 값 1일 때 실행할 명령;
					break;
		case 값2 : 변수 == 값 2일 때 실행할 명령;
					break;
		...
		
		default : 모든 case가 아닐 때 실행할 명령;
	} 
	```
